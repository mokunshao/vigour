# Slot Scope

<vigour-dropdown>
  <vigour-button>options</vigour-button>
  <template #content="{close}">
    <ul>
      <li v-for="i in 4" :key="i">
        {{ i }}
      </li>
    </ul>
    <vigour-button @click="close">close</vigour-button>
  </template>
</vigour-dropdown>
